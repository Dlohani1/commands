Basic Git Commands         Syntax

1) git clone             git clone repo_url

2) git init              git init

3) git add               git add filename
									 git add .
									 git add *				
									 git add -A   

4) git commit            git commit -m "message"

5) git push              git push origin branch_name

6) git status            git status

7) git diff              git diff filename 

8) git pull              git pull origin branch_name

9) git merge             git merge branch_name

10) git branch           git branch

		eg git branch branch_name
			git checkout branch_name


11) git checkout -b branch_name 

12) git checkout branch_name

13) git fetch --all

14) git remote -v 

15) git config --list

16) git reset  - git reset filename - will move back the added file
                from staging area to the working area

17) git checkout filename - it will undo the modified code permanently

16) git config --global user.email "youremail"
